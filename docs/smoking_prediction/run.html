165    def compute_mean_images_from_dataset(dataset, class_names, n_samples=50, img_size=(250, 250)):
166        imgs_by_class = {cls: [] for cls in class_names}
167
168        # Randomly sample subset of dataset
169        sample_indices = random.sample(range(len(dataset.samples)), min(len(dataset.samples), n_samples))
170        for idx in sample_indices:
171            path, class_idx = dataset.samples[idx]
172            try:
173                img = Image.open(path).convert("RGB").resize(img_size)
174                img = np.array(img, dtype=np.float32) / 255.0
175                imgs_by_class[class_names[class_idx]].append(img)
176            except Exception as e:
177                print(f"Skipping {path}: {e}")
178
179        mean_images = {}
180        for cls, imgs in imgs_by_class.items():
181            if imgs:
182                mean_img = np.mean(imgs, axis=0)
183                mean_img = (mean_img * 255).astype(np.uint8)
184                mean_images[cls] = Image.fromarray(mean_img)
185        
186        return mean_images

221    def load_metrics_global(ckpt_name):
222        """
223        Load training metrics for the specified checkpoint.
224
225        Returns:
226            info_text (str): Info message about checkpoint/metrics.
227            acc_text (str): Markdown string showing final validation accuracy.
228            loss_fig (plotly figure): Loss curves.
229            acc_fig (plotly figure): Accuracy curve.
230        """
231        if not ckpt_name:
232            return "No checkpoint selected", "", None, None
233        
234        metrics_path = os.path.join(BASE_DIR, "reports", "data", f"{os.path.splitext(ckpt_name)[0]}_metrics.pkl")
235        if not os.path.exists(metrics_path):
236            return f"No metrics found for {ckpt_name}", "", None, None
237
238        with open(metrics_path, "rb") as f:
239            metrics = pickle.load(f)
240
241        train_losses = metrics["train_losses"]
242        val_losses = metrics["val_losses"]
243        val_accs = metrics["val_accs"]
244        final_val_acc = val_accs[-1] if val_accs else None
245
246        min_len = min(len(train_losses), len(val_losses))
247        epochs = list(range(1, min_len + 1))
248        
249        df_loss = pd.DataFrame({
250            "Epoch": epochs + epochs,
251            "Loss": train_losses[:min_len] + val_losses[:min_len],
252            "Type": ["Train"] * min_len + ["Validation"] * min_len,
253        })
254
255        df_acc = pd.DataFrame({
256            "Epoch": list(range(1, len(val_accs) + 1)),
257            "Accuracy": val_accs,
258            "Type": ["Validation"] * len(val_accs),
259        })
260
261        loss_fig = px.line(df_loss, x="Epoch", y="Loss", color="Type", markers=True, title="Loss Curves")
262        acc_fig = px.line(df_acc, x="Epoch", y="Accuracy", color="Type", markers=True, title="Validation Accuracy")
263
264        acc_text = (
265            f"### 🏁 Final Validation Accuracy: **{final_val_acc:.2f}**"
266            if final_val_acc is not None
267            else "No accuracy data available."
268        )
269
270        return f"Metrics for {ckpt_name}", acc_text, loss_fig, acc_fig

166    def compute_mean_images_from_dataset(dataset, class_names, n_samples=50, img_size=(250, 250)):
167        imgs_by_class = {cls: [] for cls in class_names}
168
169        # Randomly sample subset of dataset
170        sample_indices = random.sample(range(len(dataset.samples)), min(len(dataset.samples), n_samples))
171        for idx in sample_indices:
172            path, class_idx = dataset.samples[idx]
173            try:
174                img = Image.open(path).convert("RGB").resize(img_size)
175                img = np.array(img, dtype=np.float32) / 255.0
176                imgs_by_class[class_names[class_idx]].append(img)
177            except Exception as e:
178                print(f"Skipping {path}: {e}")
179
180        mean_images = {}
181        for cls, imgs in imgs_by_class.items():
182            if imgs:
183                mean_img = np.mean(imgs, axis=0)
184                mean_img = (mean_img * 255).astype(np.uint8)
185                mean_images[cls] = Image.fromarray(mean_img)
186        
187        return mean_images

222    def load_metrics_global(ckpt_name):
223        """
224        Load training metrics for the specified checkpoint.
225
226        Returns:
227            info_text (str): Info message about checkpoint/metrics.
228            acc_text (str): Markdown string showing final validation accuracy.
229            loss_fig (plotly figure): Loss curves.
230            acc_fig (plotly figure): Accuracy curve.
231        """
232        if not ckpt_name:
233            return "No checkpoint selected", "", None, None
234        
235        metrics_path = os.path.join(BASE_DIR, "reports", "data", f"{os.path.splitext(ckpt_name)[0]}_metrics.pkl")
236        if not os.path.exists(metrics_path):
237            return f"No metrics found for {ckpt_name}", "", None, None
238
239        with open(metrics_path, "rb") as f:
240            metrics = pickle.load(f)
241
242        train_losses = metrics["train_losses"]
243        val_losses = metrics["val_losses"]
244        val_accs = metrics["val_accs"]
245        final_val_acc = val_accs[-1] if val_accs else None
246
247        min_len = min(len(train_losses), len(val_losses))
248        epochs = list(range(1, min_len + 1))
249        
250        df_loss = pd.DataFrame({
251            "Epoch": epochs + epochs,
252            "Loss": train_losses[:min_len] + val_losses[:min_len],
253            "Type": ["Train"] * min_len + ["Validation"] * min_len,
254        })
255
256        df_acc = pd.DataFrame({
257            "Epoch": list(range(1, len(val_accs) + 1)),
258            "Accuracy": val_accs,
259            "Type": ["Validation"] * len(val_accs),
260        })
261
262        loss_fig = px.line(df_loss, x="Epoch", y="Loss", color="Type", markers=True, title="Loss Curves")
263        acc_fig = px.line(df_acc, x="Epoch", y="Accuracy", color="Type", markers=True, title="Validation Accuracy")
264
265        acc_text = (
266            f"### 🏁 Final Validation Accuracy: **{final_val_acc:.2f}**"
267            if final_val_acc is not None
268            else "No accuracy data available."
269        )
270
271        return f"Metrics for {ckpt_name}", acc_text, loss_fig, acc_fig

Entry point for the Smoking Prediction package.

This function sets up and launches a Gradio interface for exploring a smoking dataset. It allows users to load different models, make predictions on images, train new models, analyze data distributions, and visualize model performance.

Usage: This function is intended to be used as an entry point for the PyPI installation. Once the package has been installed through PyPI, run the gradio interface by executing "smoking-prediction" in the terminal.
